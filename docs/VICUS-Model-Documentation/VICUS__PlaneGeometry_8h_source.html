39/*! Class PlaneGeometry encapsulates the vertex data and plane type of a single plane
40 in the geometrical model. Its main purpose is the triangulation of
41 the outer polygon (optionally with holes) and the triangulation of the subsurfaces and to provide
42 said data for visualization. Also, it implements intersection tests (for picking).
43
44 Since triangulation operations may be costly, the class implements lazy evaluation. Anything setting function
45 that affects the computed triangulation just marks the PlaneGeometry as dirty. The triangulation is
46 then computed on demand the triangulation data is accessed.
47
48 Note: PlaneGeometry is a runtime-only class and not used in the VICUS::Project data model. Instead,
49 all data stored in PlaneGeometry is actually held in VICUS::Surface and its member variables.
50*/
51
52class PlaneGeometry {
53public:
54
55 /*! Struct for Holes- */
56 struct Hole {
57
58 Hole(unsigned int idObj, const VICUS::Polygon2D &hole, bool isChildSurface) :
59 m_idObject(idObj),
60 m_holeGeometry(hole),
61 m_isChildSurface(isChildSurface)
62 {}
63
64 /*! ID to corresponding vicus hole object. */
65 unsigned int m_idObject = INVALID_ID;
66 /*! Hole geometry. */
67 VICUS::Polygon2D m_holeGeometry;
68 /*! Hole is part of an child surface. */
69 bool m_isChildSurface;
70 };
71
72
73 // *** PUBLIC MEMBER FUNCTIONS ***
74
75 /*! Default constructor. */
76 PlaneGeometry() {}
77
78 /*! Initializing constructor.
79 Vertexes a, b and c must be given in counter-clockwise order, so that (b-a) x (c-a) yields the normal vector of the plane.
80 */
81 PlaneGeometry(IBKMK::Polygon2D::type_t t, const IBKMK::Vector3D & a, const IBKMK::Vector3D & b, const IBKMK::Vector3D & c);
82
83 /*! Convenience constructor, initializes plane geometry with Polygon3D. */
84 PlaneGeometry(const IBKMK::Polygon3D & poly3D);
85
86 /*! A polygon is considered "fully valid" for painting, if it has enough vertexes, a valid polygon and can be
87 correctly triangulated (triangles not empty).
88 */
89 bool isValid() const;
90
91 /*! Return the inclination in Deg. 0° -> Roof; 90° -> Wall; 180° -> Floor. */
92 double inclination(int digits = 1) const;
93
94 /*! Return the orientation in Deg. 0° -> North; 90° -> East; 180° -> South; etc. */
95 double orientation(int digits = 1) const;
96
97 // Convenience query functions, using Polygon3D for data delivery
98
99 const VICUS::Polygon3D & polygon3D() const { return m_polygon; }
100 /*! Returns the offset point (origin of the plane's local coordinate system) */
101 const IBKMK::Vector3D & offset() const { return m_polygon.vertexes()[0]; }
102 const IBKMK::Vector3D & normal() const { return m_polygon.normal(); }
103 const IBKMK::Vector3D & localX() const { return m_polygon.localX(); }
104 const IBKMK::Vector3D & localY() const { return m_polygon.localY(); }
105
106 /*! Calculates surface area in m2.
107 Requires valid polygon, otherwise an exception is thrown.
108 */
109 double area(int digits = 1) const;
110
111 /*! Calculates the center point of the surface/polygon .
112 Requires valid polygon, otherwise an exception is thrown.
113 */
114 IBKMK::Vector3D centerPoint() const { return m_polygon.centerPoint(); }
115
116 // Getter functions for stored geometry
117
118 /*! Returns the 2D polygon (only if it exists) in the plane of the polygon. */
119 const IBKMK::Polygon2D & polygon2D() const { return m_polygon.polyline(); }
120 /*! Returns the vector of holes (2D polygons in the plane of the polygon). */
121 const std::vector<Hole> & holes() const { return m_holes; }
122
123 // Setter functions for plane geometry
124
125 /*! Sets the outer polygon alone. */
126 void setPolygon(const Polygon3D & polygon3D);
127
128 /*! Sets the vector of holes (2D polygons in the plane of the polygon). */
129 void setHoles(const std::vector<Hole> &holes);
130
131 /*! Set outer polygon and holes together (needs only one call to triangulate()). */
132 void setGeometry(const Polygon3D & polygon3D, const std::vector<Hole> &holes);
133
134 /*! Flips the normal vector of the geometry. Does not require update of triangulation.
135 Requires valid polygon, otherwise an exception is thrown.
136 */
137 void flip();
138
139 void changeOrigin(unsigned int idx);
140
141 // Getter functions for triangulation data (lazy evaluation)
142
143 /*! Returns the triangulation data for the opaque surface (excluding holes). */
144 const PlaneTriangulationData & triangulationData() const;
145 /*! Returns the triangulation data for the opaque surface (including holes, as if there were no holes). */
146 const PlaneTriangulationData & triangulationDataWithoutHoles() const;
147 /*! Returns the triangulation data for each of the holes. */
148 const std::vector<PlaneTriangulationData> & holeTriangulationData() const;
149
150 // Utility functions
151
152 /*! Tests if a line (with equation p = p1 + t * d) hits this plane. Returns true if
153 intersection is found, and returns the normalized distance (t) between intersection point
154 'intersectionPoint' and point p1.
155
156 The optional argument hitBackfacingPlanes disables the front-facing check (if true).
157 The optional argument endlessPlane disables the check if the intersection point
158 lies within the plane (useful for getting intersections with, for example, the xy-plane).
159
160 If plane contains holes, 'holeIndex' contains the index of the respective hole. If the plane
161 does not have any holes or the opaque surface was clicked, holeIndex will be -1.
162 */
163 bool intersectsLine(const IBKMK::Vector3D & p1,
164 const IBKMK::Vector3D & d,
165 IBKMK::Vector3D & intersectionPoint,
166 double & dist, int & holeIndex,
167 bool hitBackfacingPlanes = false,
168 bool endlessPlane = false) const;
169
170private:
171
172 // *** PRIVATE MEMBER FUNCTIONS ***
173
174 /*! This function triangulates the geometry and populate the m_triangles and m_triangleVertexes vectors.
175 This function is called whenever the polygon or the holes change.
176
177 \note This function is const, because it is called from const functions.
178 */
179 void triangulate() const;
180
181
182 // *** PRIVATE MEMBER VARIABLES ***
183
184 /*! Contains the information about the polygon that encloses this surface. */
185 Polygon3D m_polygon;
186
187 /*! Polygons with holes/subsurfaces inside the polygon. */
188 std::vector<Hole> m_holes;
189
190 /*! Contains the vertex indexes for each triangle that the polygon is composed of.
191 Includes only the triangles of the opaque surfaces without any holes
192 */
193 mutable PlaneTriangulationData m_triangulationData;
194
195 /*! Contains the vertex indexes for each triangle that the polygon is composed of.
196 Includes only the triangles of the opaque surfaces without any holes
197 */
198 mutable PlaneTriangulationData m_triangulationDataWithoutHoles;
199
200 /*! Contains the triangulation data of each hole.
201 Invalid hole definitions will yield empty triangle vectors.
202 */
203 mutable std::vector<PlaneTriangulationData> m_holeTriangulationData;
204
205 /*! Dirty flag marks the triangulation data as outdated and is set whenever plane geometry data is modified. */
206 mutable bool m_dirty = true;
207};
