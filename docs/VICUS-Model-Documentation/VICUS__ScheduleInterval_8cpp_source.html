34bool ScheduleInterval::isValid() const {
35 // check that interval start day is in range 0..364
36 if (m_intervalStartDay>364)
37 return false;
38
39 // check that the daily cycles define indeed all days
40
41 if(m_dailyCycles.empty())
42 return false;
43
44 // we need to check two things:
45 // - all day times must be parametrized
46 // - there must not be duplicate day type definitions
47
48 std::set<int> parametrizedDayTypes;
49 // process all daily cycles
50 for (const DailyCycle & dC : m_dailyCycles) {
51 // check if DailyCycle itself is correct (contains correct day types)
52 if (!dC.isValid())
53 return false;
54
55 // check if any of the parametrized day types has previously been defined
56 // and remember day typers
57
58 for ( int dT : dC.m_dayTypes) {
59 if (parametrizedDayTypes.find(dT) != parametrizedDayTypes.end())
60 return false; // already in set, duplicate!
61
62 parametrizedDayTypes.insert(dT);
63 }
64 }
65 // Holidays are optional, so we just insert this day type
66 parametrizedDayTypes.insert(NANDRAD::Schedule::ST_HOLIDAY);
67
68 // now remove all days that we require
69 std::set<int> allDayTypes;
70 allDayTypes.insert(NANDRAD::Schedule::ST_MONDAY);
71 allDayTypes.insert(NANDRAD::Schedule::ST_TUESDAY);
72 allDayTypes.insert(NANDRAD::Schedule::ST_WEDNESDAY);
73 allDayTypes.insert(NANDRAD::Schedule::ST_THURSDAY);
74 allDayTypes.insert(NANDRAD::Schedule::ST_FRIDAY);
75 allDayTypes.insert(NANDRAD::Schedule::ST_SATURDAY);
76 allDayTypes.insert(NANDRAD::Schedule::ST_SUNDAY);
77 allDayTypes.insert(NANDRAD::Schedule::ST_HOLIDAY);
78
79 // both sets must contain all required day types
80 if (allDayTypes != parametrizedDayTypes)
81 return false;
82
83 return true;
84}

87bool ScheduleInterval::usedAllDayTypes(bool withHolyday) const {
88 //set for all day types
89 std::set<int> allDayTypes;
90
91 for(const DailyCycle & dc : m_dailyCycles){
92 //save size of set for compare later
93 unsigned int dtSizeBefore = allDayTypes.size();
94 //add all current day types to set
95 for(int dt : dc.m_dayTypes)
96 allDayTypes.insert(dt);
97 //check if a day type was already in this set
98 if(dtSizeBefore == allDayTypes.size() &&dc.m_dayTypes.size()>0){
99
100 }
101 }
102
103 //create a set of all valid day types
104 std::set<int> allValidDayTypes;
105 allValidDayTypes.insert(NANDRAD::Schedule::ST_MONDAY);
106 allValidDayTypes.insert(NANDRAD::Schedule::ST_TUESDAY);
107 allValidDayTypes.insert(NANDRAD::Schedule::ST_WEDNESDAY);
108 allValidDayTypes.insert(NANDRAD::Schedule::ST_THURSDAY);
109 allValidDayTypes.insert(NANDRAD::Schedule::ST_FRIDAY);
110 allValidDayTypes.insert(NANDRAD::Schedule::ST_SATURDAY);
111 allValidDayTypes.insert(NANDRAD::Schedule::ST_SUNDAY);
112 if(withHolyday)
113 allValidDayTypes.insert(NANDRAD::Schedule::ST_HOLIDAY);
114
115 //compare the two sets
116 return allDayTypes == allValidDayTypes;
117}

128 allValidDayTypes.insert(NANDRAD::Schedule::ST_SUNDAY);
129 allValidDayTypes.insert(NANDRAD::Schedule::ST_HOLIDAY);
130
131 for(const DailyCycle & dc : m_dailyCycles){
132 //delete all current day types in set
133 for(int dt : dc.m_dayTypes){
134 if(allValidDayTypes.find(dt) != allValidDayTypes.end())
135 allValidDayTypes.erase(dt);
136 }
137 }
138 return allValidDayTypes;

142ScheduleInterval ScheduleInterval::multiply(const ScheduleInterval &other, unsigned int startDay) const {
143 FUNCID(ScheduleInterval::multiply);
144 ScheduleInterval schedInt;
145 if (!isValid())
146 IBK::Exception(IBK::FormatString("Invalid schedule interval."), FUNC_ID);
147
148 if (!other.isValid())
149 IBK::Exception(IBK::FormatString("Invalid schedule interval."), FUNC_ID);
150
151
152 //multiply each daily cycle (dc) of the first dc with the second dc
153 for(unsigned int i=0; i<m_dailyCycles.size(); ++i){
154 for(unsigned int j=0; j<other.m_dailyCycles.size();++j){
155 DailyCycle dc = m_dailyCycles[i] * other.m_dailyCycles[j];
156 if(dc != DailyCycle())
157 schedInt.m_dailyCycles.push_back(dc);
158 }
159 }
160 if(schedInt.m_dailyCycles.empty()){
161 return ScheduleInterval();
162 }
163
164 schedInt.m_displayName = m_displayName;
165 schedInt.m_intervalStartDay = startDay;
166
167 return schedInt;
168}

249void ScheduleInterval::createWeekDataVector(std::vector<double> &timepoints, std::vector<double> &data) const {
250
251 std::vector<NANDRAD::Schedule::ScheduledDayType> dts{
252 NANDRAD::Schedule::ST_MONDAY,
253 NANDRAD::Schedule::ST_TUESDAY,
254 NANDRAD::Schedule::ST_WEDNESDAY,
255 NANDRAD::Schedule::ST_THURSDAY,
256 NANDRAD::Schedule::ST_FRIDAY,
257 NANDRAD::Schedule::ST_SATURDAY,
258 NANDRAD::Schedule::ST_SUNDAY
259 };
260 for(NANDRAD::Schedule::ScheduledDayType dt1 : dts){
261 for(unsigned int i = 0; i<m_dailyCycles.size(); ++i){
262 const DailyCycle &dc = m_dailyCycles[i];
263 for(unsigned int j = 0; j<dc.m_dayTypes.size(); ++j){
264 NANDRAD::Schedule::ScheduledDayType dt = (NANDRAD::Schedule::ScheduledDayType)dc.m_dayTypes[j];
265 if(dt == dt1){
266 double addTime = 0;
267 switch (dt1) {
268 case NANDRAD::Schedule::ST_ALLDAYS:
269 case NANDRAD::Schedule::ST_WEEKDAY:
270 case NANDRAD::Schedule::ST_WEEKEND:
271 //not used
272 break;
273 case NANDRAD::Schedule::ST_MONDAY:
274 // is 0
275 break;
276 case NANDRAD::Schedule::ST_TUESDAY:
277 addTime +=24*1;
278 break;
279 case NANDRAD::Schedule::ST_WEDNESDAY:
280 addTime +=24*2;
281 break;
282 case NANDRAD::Schedule::ST_THURSDAY:
283 addTime +=24*3;
284 break;
285 case NANDRAD::Schedule::ST_FRIDAY:
286 addTime +=24*4;
287 break;
288 case NANDRAD::Schedule::ST_SATURDAY:
289 addTime +=24*5;
290 break;
291 case NANDRAD::Schedule::ST_SUNDAY:
292 addTime +=24*6;
293 break;
294 case NANDRAD::Schedule::ST_HOLIDAY:
295 //not taken into account
296 break;
297 case NANDRAD::Schedule::NUM_ST:
298 break;
299
300 }
301 //add timepoints
302 for(double tp : dc.m_timePoints)
303 timepoints.push_back(tp+addTime);
304 //add data
305 data.insert(data.end(), dc.m_values.begin(), dc.m_values.end());
306 }
307 }
308 }
309 }
310
311 // special handling for empty daily cycles
312 if (timepoints.empty()) {
313 timepoints.push_back(0);
314 data.push_back(0);
315 }
316
317 // now the data vector contains all data for a week starting at monday
318}
